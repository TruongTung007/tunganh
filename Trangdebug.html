function deleteRow(index) {

  if (confirm) {

2. Đoạn mã đã sửa lỗi hoàn chỉnh 

function deleteRow(index) {
  // CHÚ THÍCH LỖI 1: Tránh đặt tên biến trùng tên với hàm confirm() gốc của hệ thống.
  // Việc đặt 'let confirm = confirm(...)' gây ra lỗi xung đột đè hàm (confirm is not a function).
  let isConfirmed = confirm("Bạn có chắc muốn xóa?");
  
  if (isConfirmed) {
    // CHÚ THÍCH LỖI 2: Hàm .splice(index, số_lượng_cần_xóa) đang truyền tham số thứ hai là 0.
    // Số 0 nghĩa là không xóa phần tử nào cả. Ta phải đổi thành 1 để xóa đúng bản ghi tại vị trí index.
    listData.splice(index, 1);
    
    renderTable(); // Gọi lại hàm cập nhật lại giao diện bảng
  }
}
